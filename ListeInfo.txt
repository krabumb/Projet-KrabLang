S'          -> ALGORITHME S
S           -> DEBUT BLOCK FIN
BLOCK       -> BLOCKINFO BLOCK | €
BLOCKINFO   -> DECLARATION | ASSIGNATION | SI | FUNCTIONEXE
DECLARATION -> TYPE DECLAPOSS
DECLAPOSS   -> ASSIGNATION | TVF EOI
ASSIGNATION -> TVF EGAL EXPR EOI
SI          -> SI EXPR BLOCK SINONOUPAS FSI
SINONOUPAS  -> SINON BLOCK | €
TQ          -> TANTQUE EXPR BLOCK FTANT
FUNCTIONEXE -> TVF PARENTHESEO ARGUMENT PARENTHESEF EOI
ARGUMENT    -> ARGUMENTN | €
ARGUMENTN   -> ARGUMENTF ARGUMENTG
ARGUMENTF   -> EXPR | TVF
ARGUMENTG   -> VIRGULE ARGUMENTN | €
EXPR        -> PARENTHESEO EXPR PARENTHESEF | TOTPERATION EXPR | OPERAND
TOTPERATION -> EXPR BINOPERATOR | UNAOPERATOR
TYPE        -> TYPEPRIM
BINOPERATOR -> OPERATOR | OPEGAL EGAL
OPERAND     -> NUMBER | TVF | NULL | BOOLEAN | STRING
NUMBER      -> ENTIER | DECIMAL
UNAOPERATOR -> NON | MINUS | PLUS
OPEGAL      -> NON | INFSUP | EGAL
OPERATOR    -> PLUS | MINUS | COMPMATHOP | BOOLOP | INFSUP

ALGORITHME  -> "Algorithme"
DEBUT       -> "debut"
FIN         -> "fin"
EGAL        -> "="
BOOLEAN     -> "true" | "false"
TYPEPRIM    -> "entier" | "decimal" | "chaine" | "booleen"
BOOLOP      -> "et" | "ou"
COMPMATHOP  -> "/" | "*" | "%"
INFSUP      -> "<" | ">"
NULL        -> "null"
MINUS       -> "-"
PLUS        -> "+"
NON         -> "non"
PARENTHESEO -> "("
PARENTHESEF -> ")"
EOI         -> ";"
VIRGULE     -> ","
SI          -> "si"
SINON       -> "sinon"
FSI         -> "fsi"
TANTQUE     -> "tantque"
FTANT       -> "ftant"

ENTIER      -> [0-9]+
DECIMAL     -> [0-9]*\.[0-9]+
TVF         -> [A-Za-z][A-Za-z0-9_]*
STRING      -> \".*\"
